\qs{}{
    The average value of a function $f(x, y, z)$ over a solid region $E$ is defined to be 
    \[
    f_{\text{ave}} = \frac{1}{V(E)} \iiint_E f(x, y, z) \, dV,
    \]
    where $V(E)$ is the volume of $E$. Find the average value of the function $f(x, y, z) = x^2 z + y^2 z$ over the region enclosed by the paraboloid $z = 1 - x^2 - y^2$ and the plane $z = 0$.
}

}

\qs{}{
    \begin{enumerate}
        \item Evaluate the integral by changing to cylindrical coordinates:
        \[
        \int_{-1}^{1} \int_{|x|}^{\sqrt{2 - x^2}} \int_{0}^{x^2 + y^2} \sqrt{x^2 + y^2} \, dz \, dy \, dx.
        \]
        \item Evaluate the integral by changing to spherical coordinates:
        \[
        \int_{-a}^{0} \int_{0}^{\sqrt{a^2 - y^2}} \int_{-\sqrt{a^2 - x^2 - y^2}}^{\sqrt{a^2 - x^2 - y^2}} (xz^2 + yz^2) \, dz \, dx \, dy.
        \]
    \end{enumerate}
}
