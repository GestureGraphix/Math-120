\qs{}{Find the equation of the sphere for which the line segment between the points 
	$A(1,1,1)$ and $B(3,-7,3)$ is a diameter. (This means that that $A$ and $B$ are antipodal points on the sphere.)
}

\sol{
	\\
	Center of sphere: 
		\begin{center}
			\[ \frac{x_{1} + x_{2}}{2}, \frac{y_{1} + y_{2}}{2},  \frac{z_{1} + z_{2}}{2}  = \frac{1 + 3}{2},  \frac{1 + -7}{2}, \frac{1 + 3}{2} = (2, -3, 2) \] 
		\end{center}
	Dimaeter: 
		\begin{center}
			\[ \sqrt{(3-1)^{2} + (-7 - 1)^{2} + (3-1)^{2}} =  6\sqrt{2} \] 
		\end{center}
	Radius: 
		\begin{center}
			\[ r = \frac{6\sqrt{2}}{2} = 3\sqrt{2}\] 
		\end{center}
	Equation of sphere: 
		\begin{center}
			\[ (x-2)^{2} + (y+3)^{2} + (z-2)^{2} = (3\sqrt{2})^{2}\] 
			\[ (x-2)^{2} + (y+3)^{2} + (z-2)^{2} = 18 \] 
		\end{center}
}
